update လုပ်ပေးရမှာဖြစ်ပါတယ်။ အဲ့မှာဆိုရင် condition ထဲက code ဟာ word array တစ်ခုကို ပေးမှာဖြစ်ပါတယ်။ ဒါပေမဲ့ သူဟာ zero index ကနေ စတာဖြစ်တဲ့အတွက်ကြောင့် ကျွန်တော်တို့ အသုံးပြုချင်တဲ့ keys ကို specify လုပ်ဖို့ရန်အတွက် array_combine ကိုအသုံးပြုနိုင်ပါတယ်။ အဲ့မှာဆိုရင် first parameter ဟာ keys ဖြစ်ပါတယ်။ Second parameter ကတော့ values တွေပဲဖြစ်ပါတယ်။ နောက် ကျွန်တော်တို့ရဲ့ story page ကို redirect လုပ်ပေးရမှာဖြစ်ပါယတယ်။ နောက်ထပ် ကျွန်တော်တု့ိဟာ cookie ကို delete လုပ်ဖို့ရန်အတွက်ပဲဖြစ်ပါတယ်။ အဲ့ဒါအတွက် cookie file ထဲမှာ delete အတွက် another else if တစ်ခုကို add လုပ်နိုင်ပါတယ်။ Cookie ကို delete လုပ်ဖို့ရန်အတွက် past မှာ expires ဖြစ်နေတဲ့ empty string တစ်ခုအနေနဲ့ cookie ကို set လုပ်ပေးရမှာဖြစ်ပါတယ်။ နောက်ဆုံးကတော့ index.php ထဲမှာ delete အတွက် link တစ်ခုကို add လုပ်ပေးရမှာဖြစ်ပါတယ်။

Code Example

        
            <?php
            session_start();
            if (isset($_GET['save'])){
                $name = urlencode($_SESSION['word'][2]) . time();
                setcookie($name, implode(':', $_SESSION['word']), strtotime('+30 days'), '/');
            
            } else if(isset($_GET['read'])){
                $_SESSION['word'] = array_combine(range(1,5), explode(':', $_COOKIE[$_GET['read']]));
                header('location: ../js/story.php');
                exit;
            } else if (isset($_GET['delete'])){
                setcookie($_GET['delete'], '', time() - 3600, '/');
            }
            header('location:../index.php');
            exit;
            ?>
